#!/usr/bin/env python3
import os
import re
import matplotlib.pyplot as plt
import matplotlib.ticker as mticker
from tqdm import tqdm
from collections import defaultdict

directory_path = "/home/kali/Desktop/GIS/PRAC1/n"
separators = r"[:;| ]"

# Diccionario para contar la cantidad total de cada d칤gito
digit_counts = defaultdict(int)

# Obtener lista de archivos
all_files = [f for f in os.listdir(directory_path) if os.path.isfile(os.path.join(directory_path, f))]

# Contador total de caracteres procesados
total_chars = 0  

# Procesar archivos con tqdm para mostrar progreso
for filename in tqdm(all_files, desc="Processing files", unit="file"):
    file_path = os.path.join(directory_path, filename)
    
    with open(file_path, "r", encoding="utf-8", errors="ignore") as file:
        for line in file:
            parts = re.split(separators, line, maxsplit=1)
            if len(parts) == 2:
                contrasenya = parts[1].strip()
                
                if ".txt" in contrasenya or not contrasenya.isprintable():
                    continue  # Ignorar contrase침as inv치lidas

                # Contar la cantidad total de cada d칤gito
                for char in contrasenya:
                    if char in "0123456789":  
                        digit_counts[char] += 1
                        total_chars += 1  # Contamos los caracteres procesados

# Ordenar los d칤gitos del 0 al 9
digits = sorted(digit_counts.keys())  
counts = [digit_counts[d] for d in digits]

# Mostrar progreso en la terminal
print("\nNombre total de car맊ters processats:", total_chars)
for d in digits:
    print(f"D칤git {d}: {digit_counts[d]} vegades")

# Crear el gr치fico
plt.bar(digits, counts, color='skyblue')

# Agregar t칤tulos y etiquetas
plt.title("Nombre de vegades que apareix cada d칤git a les contrasenyes")
plt.xlabel("D칤git")
plt.ylabel("Nombre d'aparicions")

# 游댳 Formatear el eje Y para evitar notaci칩n cient칤fica
plt.gca().yaxis.set_major_formatter(mticker.StrMethodFormatter('{x:,.0f}'))

# Mostrar el gr치fico
plt.show()
